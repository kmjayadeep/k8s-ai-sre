  PORTKEY_API_KEY   (or MODEL_API_KEY — one required)
  MODEL_NAME

  MODEL_BASE_URL          (default: https://api.portkey.ai/v1)

  TELEGRAM_CHAT_ID
  TELEGRAM_ALLOWED_CHAT_IDS
